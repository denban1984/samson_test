CREATE TABLE a_product ( 
 ID INT(9) PRIMARY KEY AUTO_INCREMENT NOT NULL,
 PRODUCT_CODE INT(9) UNIQUE KEY NOT NULL,
 PRODUCT_NAME VARCHAR(255) NOT NULL
);

CREATE TABLE a_property (
 ID INT(9) PRIMARY KEY AUTO_INCREMENT NOT NULL,
 PRODUCT_CODE INT(9) NOT NULL,
 PRODUCT_PROPERTY VARCHAR(255) NOT NULL,
 PROPERTY_VALUE VARCHAR(255) UNIQUE KEY NOT NULL,
 FOREIGN KEY (PRODUCT_CODE) REFERENCES a_product (PRODUCT_CODE) ON DELETE CASCADE
);
